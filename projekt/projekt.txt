from openpyxl import Workbook # otwacie biblotek 
 
from openpyxl.drawing.image import Image

wb = Workbook() 
 
sheet = wb.active # aktywny arkusz 

sheet["A1"] = 'dane' # dodanie opisu w A1 w exel

n = input("wprowadź liczbę elementóW w liscie i nacisnij Enter ") # operator okresla liczbe danych 

a = int(n)

lista = []

nazwa =[]

for i in range(a):
    
    print("podaj daną", i+1)
    
    b = input ()
 
    print("podaj nazwe", i+1)
    
    q = input ()
  
   
    lista.append(b)
    
    nazwa.append(q)
    
    


pary = zip(nazwa, lista)  #sparowanie dany do sortowania 
slownik = dict(pary)  # utworzenie słownika 


S = sorted(slownik.items(), key=lambda x: x[1]) # powstaje posortwania lista 
S1 = dict(S) # ponownie słownik 

wada = str(S1.keys()) # koljeno wady od najmnieszej wartosci 
nazwa2 = S1.keys()
wartosc = str(S1.values()) # odpowiednie wartosci 
lista2 = S1.values()

sheet["A3"] = wada # lista wad
sheet["A4"] = wartosc # lista wartosc odpowidnych do wad

# utworzenie wykresu  obu list i zapis do jpg 
import matplotlib.pyplot as pyplot #bibloteka do wykresu 

pyplot.figure(figsize=((20,16)))

pyplot.bar(nazwa2, lista2, label = 'Bar 2', color = 'red',  width = 1)

pyplot.xlabel('nazwa', fontsize = 16)


pyplot.ylabel('lista', fontsize = 16)


pyplot.savefig('test.png') #wykres jpeg

pyplot.show()

img = Image('test.png')
sheet.add_image(img, 'D1')

wb.save("test.xlsx") # zapis pliku na dysku
